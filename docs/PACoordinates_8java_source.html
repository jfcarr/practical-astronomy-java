522 public HeliographicCoordinates heliographicCoordinates(double helioPositionAngleDeg, double helioDisplacementArcmin,
523 double gwdateDay, int gwdateMonth, int gwdateYear) {
524 double julianDateDays = PAMacros.civilDateToJulianDate(gwdateDay, gwdateMonth, gwdateYear);
525 double tCenturies = (julianDateDays - 2415020) / 36525;
526 double longAscNodeDeg = PAMacros.degreesMinutesSecondsToDecimalDegrees(74, 22, 0) + (84 * tCenturies / 60);
527 double sunLongDeg = PAMacros.sunLong(0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdateYear);
528 double y = Math.sin(Math.toRadians(longAscNodeDeg - sunLongDeg))
529 * Math.cos(Math.toRadians(PAMacros.degreesMinutesSecondsToDecimalDegrees(7, 15, 0)));
530 double x = -Math.cos(Math.toRadians(longAscNodeDeg - sunLongDeg));
531 double aDeg = PAMacros.wToDegrees(Math.atan2(y, x));
532 double mDeg1 = 360 - (360 * (julianDateDays - 2398220) / 25.38);
533 double mDeg2 = mDeg1 - 360 * Math.floor(mDeg1 / 360);
534 double l0Deg1 = mDeg2 + aDeg;
535 double b0Rad = Math.asin(Math.sin(Math.toRadians(sunLongDeg - longAscNodeDeg))
536 * Math.sin(Math.toRadians(PAMacros.degreesMinutesSecondsToDecimalDegrees(7, 15, 0))));
537 double theta1Rad = Math.atan(-Math.cos(Math.toRadians(sunLongDeg))
538 * Math.tan(Math.toRadians(PAMacros.obliq(gwdateDay, gwdateMonth, gwdateYear))));
539 double theta2Rad = Math.atan(-Math.cos(Math.toRadians(longAscNodeDeg - sunLongDeg))
540 * Math.tan(Math.toRadians(PAMacros.degreesMinutesSecondsToDecimalDegrees(7, 15, 0))));
541 double pDeg = PAMacros.wToDegrees(theta1Rad + theta2Rad);
542 double rho1Deg = helioDisplacementArcmin / 60;
543 double rhoRad = Math.asin(2 * rho1Deg / PAMacros.sunDia(0, 0, 0, 0, 0, gwdateDay, gwdateMonth, gwdateYear))
544 - Math.toRadians(rho1Deg);
545 double bRad = Math.asin(Math.sin(b0Rad) * Math.cos(rhoRad)
546 + Math.cos(b0Rad) * Math.sin(rhoRad) * Math.cos(Math.toRadians(pDeg - helioPositionAngleDeg)));
547 double bDeg = PAMacros.wToDegrees(bRad);
548 double lDeg1 = PAMacros.wToDegrees(
549 Math.asin(Math.sin(rhoRad) * Math.sin(Math.toRadians(pDeg - helioPositionAngleDeg)) / Math.cos(bRad)))
550 + l0Deg1;
551 double lDeg2 = lDeg1 - 360 * Math.floor(lDeg1 / 360);
552
553 double helioLongDeg = PAUtil.round(lDeg2, 2);
554 double helioLatDeg = PAUtil.round(bDeg, 2);
555
556 return new HeliographicCoordinates(helioLongDeg, helioLatDeg);
557 }
